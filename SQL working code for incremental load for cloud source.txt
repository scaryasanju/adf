-- step-5 usp_UpsertToTarget — Dynamic MERGE Stored Procedure
DROP PROCEDURE IF EXISTS usp_UpsertToTarget;
GO

CREATE PROCEDURE usp_UpsertToTarget
    @StagingTable   NVARCHAR(255),
    @TargetTable    NVARCHAR(255),
    @KeyColumn      NVARCHAR(255),
    @OldWatermark   DATETIME,
    @NewWatermark   DATETIME
AS
BEGIN
    SET NOCOUNT ON;

    DECLARE @SQL        NVARCHAR(MAX);
    DECLARE @UpdateCols NVARCHAR(MAX) = '';
    DECLARE @InsertCols NVARCHAR(MAX) = '';
    DECLARE @ValuesCols NVARCHAR(MAX) = '';

    -- Build column lists dynamically
    SELECT
        @UpdateCols = @UpdateCols +
            CASE WHEN COLUMN_NAME = @KeyColumn THEN ''
                 ELSE 'TGT.[' + COLUMN_NAME + '] = SRC.[' + COLUMN_NAME + '], '
            END,
        @InsertCols = @InsertCols + '[' + COLUMN_NAME + '], ',
        @ValuesCols = @ValuesCols + 'SRC.[' + COLUMN_NAME + '], '
    FROM INFORMATION_SCHEMA.COLUMNS
    WHERE TABLE_NAME = @TargetTable
    ORDER BY ORDINAL_POSITION;

    -- Remove trailing commas
    SET @UpdateCols = LEFT(@UpdateCols, LEN(@UpdateCols) - 1);
    SET @InsertCols = LEFT(@InsertCols, LEN(@InsertCols) - 1);
    SET @ValuesCols = LEFT(@ValuesCols, LEN(@ValuesCols) - 1);

    -- Build MERGE with explicit CAST on UpdatedAt in staging
    SET @SQL = '
    MERGE INTO [' + @TargetTable + '] AS TGT
    USING (
        SELECT * FROM [' + @StagingTable + ']
        WHERE  CAST(UpdatedAt AS DATETIME) > @OldWM
          AND  CAST(UpdatedAt AS DATETIME) <= @NewWM
    ) AS SRC
    ON  TGT.[' + @KeyColumn + '] = SRC.[' + @KeyColumn + ']
    WHEN MATCHED THEN
        UPDATE SET ' + @UpdateCols + '
    WHEN NOT MATCHED BY TARGET THEN
        INSERT (' + @InsertCols + ')
        VALUES (' + @ValuesCols + ');';

    EXEC sp_executesql
        @SQL,
        N'@OldWM DATETIME, @NewWM DATETIME',
        @OldWM = @OldWatermark,
        @NewWM = @NewWatermark;

END;
GO

-- step-6 usp_UpdateWatermark — Save Watermark After Each Run

CREATE PROCEDURE usp_UpdateWatermark
    @TableName     NVARCHAR(255),
    @NewWatermark  DATETIME
AS
BEGIN
    UPDATE watermark_table
    SET    WatermarkValue = @NewWatermark
    WHERE  TableName = @TableName;
END;
